3️⃣ Capability Resolver (Single Enforcement Gate)
// /src/capabilities/canPerform.ts

export function canPerform(
  plan: Plan,
  capability: Capability,
  usage: number
): { allowed: boolean; reason?: string } {
  const rules = resolveRulesForPlan(plan, capability)

  if (!rules) return { allowed: false, reason: 'Upgrade required' }

  if (rules.unlimited) return { allowed: true }

  if (rules.limit !== undefined && usage >= rules.limit) {
    return {
      allowed: false,
      reason: 'You’ve reached the limit for your plan.'
    }
  }

  if (rules.mode === 'read_only') {
    return {
      allowed: false,
      reason: 'Upgrade to take action.'
    }
  }

  return { allowed: true }
}

4️⃣ Enforcement Pattern (Inline, Non-Blocking)

Apply enforcement only when an action is attempted.

Example: Creating a Job
const result = canPerform(
  user.plan,
  'jobs.create',
  user.usage.jobs_created
)

if (!result.allowed) {
  showInlineMessage(result.reason)
  return
}

Example: Enabling Deposit Enforcement
if (!canPerform(user.plan, 'deposit.enforce', user.usage.deposits_used).allowed) {
  showInlineMessage('Deposit protection is limited on your plan.')
}
