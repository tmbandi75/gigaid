## 1️⃣ Add Dev-Only Plan Override (Safe Testing)

### Update `/src/lib/entitlements.ts`

Add support for a **development override** using an env var.

```ts
import { Plan, Capability, PLAN_CAPABILITIES } from "./plans";

const DEV_EMAILS = [
  "your@email.com"
];

export function isDeveloper(user?: { email?: string }) {
  return !!user?.email && DEV_EMAILS.includes(user.email);
}

export function getUserCapabilities(user?: {
  email?: string;
  plan?: Plan;
}) {
  // 🔑 Developer bypass (always full access)
  if (isDeveloper(user)) {
    return new Set(Object.values(PLAN_CAPABILITIES).flat());
  }

  // 🧪 Dev-only plan override
  const overridePlan =
    process.env.NODE_ENV !== "production"
      ? (process.env.DEV_PLAN_OVERRIDE as Plan | undefined)
      : undefined;

  const effectivePlan = overridePlan ?? user?.plan ?? Plan.FREE;

  return new Set(PLAN_CAPABILITIES[effectivePlan]);
}
```

⚠️ Guardrail

* `DEV_PLAN_OVERRIDE` must have **no effect in production**
