Step 3 — Ensure checkout always attempts API call

Ensure the Stripe entry function begins execution immediately:

export async function startStripeCheckout(planId: string) {
  console.log("[Stripe] Entered startStripeCheckout");

  console.log("[Stripe] Creating checkout session");

  const res = await fetch("/api/checkout/session", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ planId }),
  });

  if (!res.ok) {
    const data = await res.json().catch(() => ({}));
    toast({
      title: "Payments temporarily unavailable",
      description: data.error ?? "Please try again later.",
      variant: "destructive",
    });
    return;
  }

  const session = await res.json();
  console.log("[Stripe] Redirecting to Stripe Checkout", session.url);
  window.location.href = session.url;
}


✅ No env usage
✅ No Stripe assumptions
✅ Backend decides outcome
