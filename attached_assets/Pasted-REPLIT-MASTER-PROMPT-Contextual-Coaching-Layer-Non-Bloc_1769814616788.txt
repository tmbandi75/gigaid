Architecture
1. Coaching Message Registry

Create a centralized registry defining message → trigger → screen → conditions.

// /src/coaching/coachingMessages.ts

export type CoachingMessage = {
  id: string
  screen: 'dashboard' | 'jobs' | 'leads' | 'invoices' | 'tools'
  placement: 'header' | 'empty_state' | 'inline' | 'toast' | 'tooltip'
  text: string
  condition: (state: AppState) => boolean
  once: boolean
}

export const COACHING_MESSAGES: CoachingMessage[] = [
  {
    id: 'dashboard_add_service',

2. Coaching Display State

Persist which messages have been shown.

// /src/state/useCoachingState.ts

export const useCoachingState = () => {
  const [seen, setSeen] = useLocalStorage<string[]>('coaching_seen', [])

  const markSeen = (id: string) => {
    if (!seen.includes(id)) {
      setSeen([...seen, id])
    }
  }

  const hasSeen = (id: string) => seen.includes(id)

  return { hasSeen, markSeen }
}

3. Coaching Renderer (Non-Blocking)
// /src/coaching/CoachingRenderer.tsx

export function CoachingRenderer({ screen }: { screen: Screen }) {
  const { hasSeen, markSeen } = useCoachingState()
  const appState = useAppState()

  const message = COACHING_MESSAGES.find(
    m =>
      m.screen === screen &&
      m.condition(appState) &&
      (!m.once || !hasSeen(m.id))
  )

  if (!message) return null

  useEffect(() => {
    markSeen(message.id)
  }, [])

  return (
    <div className="coaching-message">
      {message.text}
    </div>
  )
}
