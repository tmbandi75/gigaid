## 3️⃣ Developer Bypass (Critical)

### Create:

`/src/lib/entitlements.ts`

```ts
import { Plan, Capability, PLAN_CAPABILITIES } from "./plans";

const DEV_EMAILS = [
  "your@email.com"
];

export function isDeveloper(user?: { email?: string }) {
  return !!user?.email && DEV_EMAILS.includes(user.email);
}

export function getUserCapabilities(user?: {
  email?: string;
  plan?: Plan;
}): Set<Capability> {

  // 🔑 Developer always has full access
  if (isDeveloper(user)) {
    return new Set(
      Object.values(PLAN_CAPABILITIES).flat()
    );
  }

  const plan = user?.plan ?? Plan.FREE;
  return new Set(PLAN_CAPABILITIES[plan]);
}
```

> ✅ You will **never** see upgrade prompts
> ✅ You can test everything
> ✅ No fake subscriptions needed

## 4️⃣ Capability Check Helper

### Create:

`/src/lib/hasCapability.ts`

```ts
import { Capability } from "./plans";
import { getUserCapabilities } from "./entitlements";

export function hasCapability(
  user: any,
  capability: Capability
): boolean {
  return getUserCapabilities(user).has(capability);
}
```
